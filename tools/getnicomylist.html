GetNicoMyList

ニコニコ動画のマイリストをブラウザから取得できるスクリプトです。
ブラウザの開発者コンソールから実行します。 
totalLists = []
vLinkList = document.getElementsByClassName("VideoMediaObjectList")[0].getElementsByClassName("NC-Link NC-MediaObject-contents")
for (i = 0; i < vLinkList.length; i++) {
    totalLists.push([vLinkList[i].href, vLinkList[i].getElementsByClassName(
    "NC-MediaObjectTitle NC-VideoMediaObject-title NC-MediaObjectTitle_fixed2Line")[0].innerText + "\n"])
}
const blob = new Blob(totalLists, { type: 'text/plain' });
const aTag = document.createElement('a');
aTag.href = URL.createObjectURL(blob);
aTag.target = '_blank';
aTag.download = "down.txt";
aTag.click();
URL.revokeObjectURL(aTag.href);
